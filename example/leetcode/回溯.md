https://leetcode-cn.com/problems/word-search/

<?php
/**
     * @param String[][] $board
     * @param String $word
     * @return Boolean
     */
    function exist($board, $word) {
        for($i = 0;$i < count($board);$i++){
            for($j = 0;$j<count($board[0]);$j++){
                // start 表示当前word的位置
                if($this->dfs($board,$i,$j,$word,0)){
                    return true;
                }
            }
        }
        return false;
    }
    function dfs($board,$i,$j,$word,$start){
        if($i < 0 || $j < 0 || $i > count($board)-1 || $j> count($board[0])-1 ){
            return false;
        }
        if($board[$i][$j] != $word[$start]){
            return false;
        }
        if($start == strlen($word)-1){
            return true;
        }
        // 将用过的值给覆盖掉，避免重复使用
        $board[$i][$j] = '#';
        if($this->dfs($board,$i-1,$j,$word,$start+1)){
            return true;
        }
        if($this->dfs($board,$i,$j-1,$word,$start+1)){
            return true;
        }
        if($this->dfs($board,$i+1,$j,$word,$start+1)){
            return true;
        }
        if($this->dfs($board,$i,$j+1,$word,$start+1)){
            return true;
        }
        return false;
    }

78. 子集
https://leetcode-cn.com/problems/subsets/
给定一组不含重复元素的整数数组 nums，返回该数组所有可能的子集（幂集）。

<?php

class Solution {
    public $result = [];
    /**
     * @param Integer[] $nums
     * @return Integer[][]
     */
    function subsets($nums) {
        $this->result[] = [];
        $this->dfs([],0,$nums);
        return $this->result;
    }

    function dfs($list,$start,$nums){
        if (count($list) == count($nums)) {
            return;
        }
        for($i = $start;$i<count($nums);$i++){
            $list[] = $nums[$i];
            $this->result[] = $list;
            $this->dfs($list,$i+1,$nums);
            array_pop($list);
        }
    }
}

46. 全排列
给定一个 没有重复 数字的序列，返回其所有可能的全排列。
https://leetcode-cn.com/problems/permutations/
<?php

class Solution {
    public $result = [];
    /**
     * @param Integer[] $nums
     * @return Integer[][]
     */
    function permute($nums) {
        $len = count($nums);
        if($len == 0){
            return $this->result;
        }
        $this->dfs($nums,[],0);
        return $this->result;
    }
    function dfs($nums,$list,$depth){
        if($depth == count($nums)){
            $this->result[] = $list;
            return;
        }
        for($i = 0;$i<count($nums);$i++){
            if(in_array($nums[$i],$list)) continue;
            $list[] = $nums[$i];
            $this->dfs($nums,$list,$depth + 1);
            array_pop($list);
        }
    }
}

39. 组合总和
https://leetcode-cn.com/problems/combination-sum/
给定一个无重复元素的数组 candidates 和一个目标数 target ，找出 candidates 中所有可以使数字和为 target 的组合。

<?php

class Solution {
    public $result = [];
    /**
     * @param Integer[] $candidates
     * @param Integer $target
     * @return Integer[][]
     */
    function combinationSum($candidates, $target) {
        if($target <= 0){
            return [];
        }
        sort($candidates);
        $this->sort1($candidates,$target,[],0);

        return $this->result;
    }

    function sort1($candidates,$target,$list,$start){
        if($target<0)return ;
        if($target == 0){
            $this->result[] = $list; 
        }
        for($i = $start;$i < count($candidates);++$i){
             if($target-$candidates[$i] < 0) return;
             $list[] = $candidates[$i];
             $this->sort1($candidates,$target-$candidates[$i],$list,$i);
             array_pop($list);
        }

    } 
}

17. 电话号码的字母组合
https://leetcode-cn.com/problems/letter-combinations-of-a-phone-number/
<?php

class Solution {
    /**
    * 回溯法，思路求解
    *
    **/

    public $arr = [];
    public $dic = array(
                '1' => array(),
                '2' => array('a','b','c'),
                '3' => array('d','e','f'),
                '4' => array('g','h','i'),
                '5' => array('j','k','l'),
                '6' => array('m','n','o'),
                '7' => array('p','q','r','s'),
                '8' => array('t','u','v'),
                '9' => array('w','x','y','z')
        );
    /**
     * @param String $digits
     * @return String[]
     */
    function letterCombinations($digits) {
        if($digits == ""){
            return [];
        }
        $this->dfs('',0,$digits);
        return $this->arr;
    }

    function dfs($str,$index,$digits){
        // 判断是最后一个则结束
        if($index == strlen($digits)){
            $this->arr[] = $str; 
            return;
        }
        // 第几个元素
        $letter = $digits[$index];
        foreach($this->dic[$letter] as $item){
            $this->dfs($str.$item,$index+1,$digits);
        }
    }
}
